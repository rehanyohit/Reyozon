<script>
let deals = [
  {title:"Nike Air Max - 40% OFF", category:"Fashion", link:"https://nike.com"},
  {title:"PS5 Discount Deal", category:"Gaming", link:"https://amazon.com"},
  {title:"iPhone Sale Live Now", category:"Electronics", link:"https://apple.com"}
];

// LOAD POINTS
let points = localStorage.getItem("points") || 0;
document.getElementById("points").innerText = points;

// TRACK CLICKED DEALS
let clicked = JSON.parse(localStorage.getItem("clicked")) || [];

// RENDER DEALS
function renderDeals(list) {
  let container = document.getElementById("deals");
  container.innerHTML = "";

  list.forEach((deal, index) => {
    let div = document.createElement("div");
    div.className = "card";

    div.innerHTML = `
      <div class="badge">${deal.category}</div>
      <h3>${deal.title}</h3>
      <p>Tap to open deal →</p>
    `;

    div.onclick = () => openDeal(index, deal.link);

    container.appendChild(div);
  });
}

// DEAL CLICK LOGIC
function openDeal(index, link) {
  if (!clicked.includes(index)) {
    let gain = Math.floor(Math.random() * 10) + 5;
    points = parseInt(points) + gain;

    localStorage.setItem("points", points);
    document.getElementById("points").innerText = points;

    clicked.push(index);
    localStorage.setItem("clicked", JSON.stringify(clicked));
  }

  window.open(link, "_blank");
}

// FILTER
function filterDeals(cat) {
  if (cat === "All") return renderDeals(deals);
  renderDeals(deals.filter(d => d.category === cat));
}

// SEARCH
document.getElementById("search").addEventListener("input", e => {
  let val = e.target.value.toLowerCase();
  renderDeals(deals.filter(d => d.title.toLowerCase().includes(val)));
});

// AUTO REFRESH EVERY 5 DAYS
let lastUpdate = localStorage.getItem("lastUpdate");
let now = Date.now();

if (!lastUpdate || now - lastUpdate > 5 * 24 * 60 * 60 * 1000) {
  deals.push({
    title: "🔥 Fresh Limited Deal Added",
    category: "Fashion",
    link: "https://example.com"
  });

  localStorage.setItem("lastUpdate", now);
}

// INIT
renderDeals(deals);
</script>
